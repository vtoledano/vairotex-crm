const SS_ID = '1QsDOzDk93fygprOJiqsE9njMD4qmMKd4-hfHE4xpKEo'; function doGet(e) { try { const action = e.parameter.action || 'all'; const callback = e.parameter.callback; const ss = SpreadsheetApp.openById(SS_ID); let result; if (e.parameter.data) { const d = JSON.parse(decodeURIComponent(e.parameter.data)); if (action === 'nuevo_seguimiento') result = { ok: true, data: nuevoSeguimiento(ss, d) }; else if (action === 'nuevo_cliente') result = { ok: true, data: nuevoCliente(ss, d) }; else if (action === 'nuevo_producto') result = { ok: true, data: nuevoProducto(ss, d) }; else if (action === 'actualizar_estado') result = { ok: true, data: actualizarEstado(ss, d) }; else result = { ok: false, error: 'Acción no reconocida' }; const json = JSON.stringify(result); return ContentService.createTextOutput(callback ? callback + '(' + json + ')' : json) .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } if (action === 'all') { result = { ok: true, data: { clientes: getClientes(ss), seguimientos: getSeguimientos(ss), productos: getProductos(ss), ts: new Date().toISOString() }}; } else if (action === 'clientes') result = { ok: true, data: getClientes(ss) }; else if (action === 'seguimientos') result = { ok: true, data: getSeguimientos(ss) }; else if (action === 'productos') result = { ok: true, data: getProductos(ss) }; else result = { ok: false, error: 'Acción no reconocida' }; const json = JSON.stringify(result); return ContentService.createTextOutput(callback ? callback + '(' + json + ')' : json) .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } catch(err) { const cb = e.parameter && e.parameter.callback; const r = JSON.stringify({ ok: false, error: err.message }); return ContentService.createTextOutput(cb ? cb + '(' + r + ')' : r) .setMimeType(cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; const ss = SpreadsheetApp.openById(SS_ID); if (action === 'nuevo_seguimiento') return ok(nuevoSeguimiento(ss, body.data)); if (action === 'nuevo_cliente') return ok(nuevoCliente(ss, body.data)); if (action === 'actualizar_estado') return ok(actualizarEstado(ss, body.data)); if (action === 'nuevo_producto') return ok(nuevoProducto(ss, body.data)); return err('Acción no reconocida'); } catch(e) { return err(e.message); } } function ok(data) { return ContentService.createTextOutput(JSON.stringify({ ok: true, data })) .setMimeType(ContentService.MimeType.JSON); } function err(msg) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: msg })) .setMimeType(ContentService.MimeType.JSON); } function getClientes(ss) { const sh = ss.getSheetByName('CLIENTES'); const data = sh.getDataRange().getValues(); const headers = data[0]; return data.slice(1) .filter(r => r[4] && String(r[4]).trim() !== '') .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i] === null ? '' : r[i]); return obj; }); } function getSeguimientos(ss) { const sh = ss.getSheetByName('SEGUIMIENTOS'); const data = sh.getDataRange().getValues(); const headers = data[0]; return data.slice(1) .filter(r => r[4] && String(r[4]).trim() !== '') .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i] === null ? '' : r[i]); return obj; }); } function getProductos(ss) { const sh = ss.getSheetByName('PRODUCTOS'); const data = sh.getDataRange().getValues(); const headers = data[0]; return data.slice(1) .filter(r => r[1] && String(r[1]).trim() !== '') .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i] === null ? '' : r[i]); return obj; }); } function nuevoSeguimiento(ss, d) { const sh = ss.getSheetByName('SEGUIMIENTOS'); const newId = sh.getLastRow(); const fecha = d.fecha || Utilities.formatDate(new Date(), 'America/Argentina/Buenos_Aires', 'dd/MM/yyyy'); sh.appendRow([newId, fecha, d.sucursal||'', d.tipo_empresa||'', d.empresa||'', d.tipo_contacto||'', d.contacto||'', d.telefono||'', d.productos||'', d.resumen||'', d.estado||'', d.accion||'', d.proxima_accion||'']); if (d.empresa && d.estado) actualizarEstado(ss, { empresa: d.empresa, estado: d.estado, accion: d.accion }); return { id: newId, mensaje: 'Seguimiento guardado' }; } function nuevoCliente(ss, d) { const sh = ss.getSheetByName('CLIENTES'); const newId = sh.getLastRow(); sh.appendRow([newId, d.sucursal||'', d.tipo||'', d.categoria||'', d.empresa||'', d.contacto||'', d.telefono||'', d.posicion||'', d.estado||'Contactado', d.productos||'', d.vol_proyectado||'', d.detalle||'', d.accion||'', d.muestras||'', '', '']); return { id: newId, mensaje: 'Cliente ' + d.empresa + ' guardado' }; } function actualizarEstado(ss, d) { const sh = ss.getSheetByName('CLIENTES'); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][4]).toLowerCase().trim() === String(d.empresa).toLowerCase().trim()) { sh.getRange(i + 1, 9).setValue(d.estado); if (d.fecha) sh.getRange(i + 1, 15).setValue(d.fecha); if (d.accion) sh.getRange(i + 1, 13).setValue(d.accion); return { empresa: d.empresa, estado: d.estado, fila: i + 1 }; } } return { mensaje: 'Empresa no encontrada' }; } function nuevoProducto(ss, d) { const sh = ss.getSheetByName('PRODUCTOS'); const newId = sh.getLastRow(); sh.appendRow([newId, d.codigo||'', d.familia||'', d.descripcion||'', d.tipo||'', 'SÍ']); return { id: newId, mensaje: 'Producto ' + d.codigo + ' guardado' }; }
